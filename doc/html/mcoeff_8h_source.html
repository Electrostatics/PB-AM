339 inline REAL
340 CMCoeff::computeDev(const CMCoeff & M1, const CMCoeff & M2)
341 {
342  REAL sum = 0;
343 
344  assert(M1.m_p == M2.m_p);
345 
346  for (int n = 0; n < M2.m_p; n++)
347  for (int m = 0; m <= n; m++)
348  {
349  if (M1(n,m) == Complex(0.0,0.0) && M2(n,m) == Complex(0.0,0.0))
350  continue;
351 
352  Complex a;
353  if (fabs(M1(n,m).real()) < PREC_LIMIT &&
354  fabs(M1(n,m).imag()) <PREC_LIMIT)
355  a = M2(n,m);
356  else if (fabs(M2(n,m).real()) < PREC_LIMIT &&
357  fabs(M2(n,m).imag()) <PREC_LIMIT)
358  a = M1(n,m);
359  else
360  a = 0.5*(M1(n,m) - M2(n,m))/(M1(n,m) + M2(n,m));
361 
362  sum += a.real()*a.real() + a.imag()*a.imag();
363  }
364 
365  return sum;
366 }
