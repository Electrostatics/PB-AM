354 inline REAL
355 CMCoeff::computeDev(const CMCoeff & M1, const CMCoeff & M2)
356 {
357  REAL sum = 0;
358 
359  assert(M1.m_p == M2.m_p);
360 
361  for (int n = 0; n < M2.m_p; n++)
362  for (int m = 0; m <= n; m++)
363  {
364  if (M1(n,m) == Complex(0.0,0.0) && M2(n,m) == Complex(0.0,0.0))
365  continue;
366 
367  Complex a;
368  if (fabs(M1(n,m).real()) < PREC_LIMIT &&
369  fabs(M1(n,m).imag()) <PREC_LIMIT)
370  a = M2(n,m);
371  else if (fabs(M2(n,m).real()) < PREC_LIMIT &&
372  fabs(M2(n,m).imag()) <PREC_LIMIT)
373  a = M1(n,m);
374  else
375  a = 0.5*(M1(n,m) - M2(n,m))/(M1(n,m) + M2(n,m));
376 
377  sum += a.real()*a.real() + a.imag()*a.imag();
378  }
379 
380  return sum;
381 }
