245 inline REAL
246 CMCoeff::computeDev(const CMCoeff & M1, const CMCoeff & M2)
247 {
248  REAL sum = 0;
249 
250  assert(M1.m_p == M2.m_p);
251 
252  for (int n = 0; n < M2.m_p; n++)
253  for (int m = 0; m <= n; m++)
254  {
255  if (M1(n,m) == Complex(0.0,0.0) && M2(n,m) == Complex(0.0,0.0))
256  continue;
257 
258  Complex a;
259  if (fabs(M1(n,m).real()) < PREC_LIMIT &&
260  fabs(M1(n,m).imag()) <PREC_LIMIT)
261  a = M2(n,m);
262  else if (fabs(M2(n,m).real()) < PREC_LIMIT &&
263  fabs(M2(n,m).imag()) <PREC_LIMIT)
264  a = M1(n,m);
265  else
266  a = 0.5*(M1(n,m) - M2(n,m))/(M1(n,m) + M2(n,m));
267 
268  sum += a.real()*a.real() + a.imag()*a.imag();
269  }
270 
271  return sum;
