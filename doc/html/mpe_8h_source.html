326 CMPE::sphToCartMat(const CPnt & P, CPnt * R)
327 {
328  CSpPnt d = CartToSph(P);
329  REAL sin_t = sin(d.theta());
330  REAL cos_t = cos(d.theta());
331  REAL sin_p = sin(d.phi());
332  REAL cos_p = cos(d.phi());
333  REAL ir = 1/d.rho();
334  REAL irt = ir/sin_t;
335 
336  R[0] = CPnt(sin_t*cos_p, ir*cos_t*cos_p, -irt*sin_p);
337  R[1] = CPnt(sin_t*sin_p, ir*cos_t*sin_p, irt*cos_p);
338  R[2] = CPnt(cos_t, -ir*sin_t, 0.0);
339 }
