148 {

150  REAL sin_t = sin(d.theta());
151  REAL cos_t = cos(d.theta());
152  REAL sin_p = sin(d.phi());
153  REAL cos_p = cos(d.phi());
154  REAL ir = 1/d.rho();
155  REAL irt = ir/sin_t;

157  R[0] = CPnt(sin_t*cos_p, ir*cos_t*cos_p, -irt*sin_p);
158  R[1] = CPnt(sin_t*sin_p, ir*cos_t*sin_p, irt*cos_p);
159  R[2] = CPnt(cos_t, -ir*sin_t, 0.0);
