212 CMPE::sphToCartMat(const CPnt & P, CPnt * R)
213 {
214  CSpPnt d = CartToSph(P);
215  REAL sin_t = sin(d.theta());
216  REAL cos_t = cos(d.theta());
217  REAL sin_p = sin(d.phi());
218  REAL cos_p = cos(d.phi());
219  REAL ir = 1/d.rho();
220  REAL irt = ir/sin_t;
221 
222  R[0] = CPnt(sin_t*cos_p, ir*cos_t*cos_p, -irt*sin_p);
223  R[1] = CPnt(sin_t*sin_p, ir*cos_t*sin_p, irt*cos_p);
224  R[2] = CPnt(cos_t, -ir*sin_t, 0.0);
225 }
